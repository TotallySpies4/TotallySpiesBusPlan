export async function sendToKafka(topic, data) {
    const producer = kafka.producer();
    await producer.connect();
    await producer.send({
        topic,
        messages: [{ value: JSON.stringify(data) }],
    });
    console.log(`Sent to Kafka on topic: ${topic}`);

}
